# Create PostgreSQL credentials secret
kubectl create secret generic autocorp-secrets \
  --from-literal=PGPASSWORD='your_postgres_password'

# Create GCP credentials secret for Cloud SQL proxy
kubectl create secret generic gcp-creds \
  --from-file=autocorp_storage.json=autocorp_storage.json

#Verify
kubectl get secrets

#Deploy
kubectl apply -f deployment.yaml
kubectl apply -f service.yaml

# Check pods
kubectl get pods

# Check service
kubectl get svc autocorp-service

# Create Cloud SQL instance
gcloud sql instances create autocorp-db \
  --database-version=POSTGRES_15 \
  --tier=db-g1-small \
  --region=us-central1

# Create database inside instance
gcloud sql databases create autocorp --instance=autocorp-db

# Create user
gcloud sql users create autocorp_user --instance=autocorp-db --password='yourpassword'

#grant permissions to service account
PROJECT_ID=autocorp-476900
SA_EMAIL=autocorp-storage-agent@autocorp-476900.iam.gserviceaccount.com

gcloud projects add-iam-policy-binding $PROJECT_ID \
  --member="serviceAccount:$SA_EMAIL" \
  --role="roles/cloudsql.client"

gcloud projects add-iam-policy-binding $PROJECT_ID \
  --member="serviceAccount:$SA_EMAIL" \
  --role="roles/cloudsql.instanceUser"

gcloud projects add-iam-policy-binding $PROJECT_ID \
  --member="serviceAccount:$SA_EMAIL" \
  --role="roles/cloudsql.editor"

# Connect with temporary whitelist
gcloud sql connect autocorp-db --user=autocorp_user --database=autocorp

##Cloud SQL Proxy – Setup, IAM, and Kubernetes Integration (Complete Guide)
#enable cloud SQL API
gcloud services enable sqladmin.googleapis.com

#create a service account for proxy
gcloud iam service-accounts create autocorp-storage-agent \
  --description="Service account for connecting to Cloud SQL from GKE" \
  --display-name="AutoCorp Cloud SQL Proxy"

#grant required IAM roles
PROJECT_ID=autocorp-476900
SA_EMAIL=autocorp-storage-agent@${PROJECT_ID}.iam.gserviceaccount.com

# Allow connecting to Cloud SQL instances
gcloud projects add-iam-policy-binding $PROJECT_ID \
  --member="serviceAccount:$SA_EMAIL" \
  --role="roles/cloudsql.client"

# Allow acting as SQL instance user
gcloud projects add-iam-policy-binding $PROJECT_ID \
  --member="serviceAccount:$SA_EMAIL" \
  --role="roles/cloudsql.instanceUser"

# Optional but useful for admin-level debugging
gcloud projects add-iam-policy-binding $PROJECT_ID \
  --member="serviceAccount:$SA_EMAIL" \
  --role="roles/cloudsql.editor"


#download the service account key (json)
gcloud iam service-accounts keys create autocorp_storage.json \
  --iam-account=$SA_EMAIL

#create Kubernetes secret with credentials
kubectl create secret generic gcp-creds \
  --from-file=autocorp_storage.json=autocorp_storage.json

#verify:
kubectl get secrets

#Add Cloud SQL Proxy as a Sidecar Container in Deployment
Add this block to your deployment.yaml:

- name: cloudsql-proxy
  image: gcr.io/cloud-sql-connectors/cloud-sql-proxy:2.8.0
  args:
    - "autocorp-476900:us-central1:autocorp-db"
    - "--credentials-file=/secrets/autocorp_storage.json"
  securityContext:
    runAsNonRoot: true

--------------Mount the secret volume: ---------

volumes:
  - name: gcp-creds
    secret:
      secretName: gcp-creds

---------------  Mount it inside both containers: -------------

volumeMounts:
  - name: gcp-creds
    mountPath: /secrets
    readOnly: true

------------------- Ensure your app connects via localhost:------------

- name: PGHOST
  value: "127.0.0.1"
